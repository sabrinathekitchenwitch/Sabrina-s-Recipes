\begin{multicols}{2}
\textbf{Ingredients}
\begin{itemize}
\item $1\frac{1}{2}$ white sugar 
\item 1 (7 ounce) jar marshmallow creme
\item $\frac{2}{3}$ cup evaporated milk
\item $\frac{1}{4}$ cup unsalted butter
\item $\frac{1}{4}$ tsp. kosher salt
\item 2 cups milk chocolate chips
\item 1 cup semisweet chocolate chips
\item 1 teaspoon vanilla extract
\item $\frac{1}{2}$ cup chopped nuts (optional) 


\end{itemize}


\columnbreak
\textbf{Procedure:}
\medskip


\begin{enumerate}
\item Line an 8-inch square pan with aluminum foil; set aside.
\item Place spices in water and bring to boil. Simmer for 20-25 minutes. 

\item Combine sugar, marshmallow cream, evaporated milk, butter, and salt together in a large saucepan over medium heat; bring to a full boil and cook for 5 minutes, stirring constantly.

\item Remove from heat and add milk chocolate chips and semisweet chocolate chips; stir until chocolate is melted and mixture is smooth. Stir in nuts and vanilla.

\item Pour into prepared pan; chill in refrigerator for 2 hours, or until firm.


 
\end{enumerate}
