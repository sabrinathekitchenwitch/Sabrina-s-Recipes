\begin{centering}
    \Huge{\textbf{Chicken Marsala}}
    
\end{centering}


\begin{multicols}{2}
\textbf{Ingredients}
\begin{itemize}
\item 1 $\frac{1}{2}$ pounds boneless skinless chicken breasts, pounded $\frac{1}{4}$-inch thick

\item 3 tbsp. all-purpose flour 
\item 3 tbsp. unsalted butter, divided
\item $\frac{1}{2}$ lb baby bella mushrooms 
\item 1 shallot, minced 
\item 2 cloves garlic, minced 
\item $\frac{2}{3}$ cup chicken broth
\item $\frac{2}{3}$ cup dry white wine
\item $\frac{2}{3}$ cup heavy cream
\item 2 tsp. chopped fresh thyme
\item 2 tbsp. chopped fresh Italian parsley (optional) 
\item 1 tbsp. olive oil



\end{itemize}


\columnbreak
\textbf{Procedure:}
\medskip


\begin{enumerate}

\item Place the flour, $\frac{3}{4}$ teaspoon salt, and $\frac{1}{4}$ teaspoon pepper in a ziplock bag. Add the chicken to the bag; seal bag tightly and shake to coat chicken evenly. Set aside.
\medskip

\item Heat the oil and 2 tablespoons of the butter in a large skillet over medium-high heat. Place the flour-dusted chicken in the pan, shaking off any excess first, and cook, turning once, until the chicken is golden and just barely cooked through, about 5 to 6 minutes total. Transfer the chicken to a plate and set aside.
\medskip 
\item Melt the remaining tablespoon of butter in the pan. Add the mushrooms and cook, stirring frequently, until the mushrooms begin to brown, 3 to 4 minutes. Add the shallots, garlic, and $\frac{1}{4}$ teaspoon of salt; cook for 1 to 2 minutes more. Add the broth, wine, heavy cream, thyme, $\frac{1}{4}$ teaspoon salt, and $\frac{1}{4}$ teaspoon of pepper; use a wooden spoon to scrape any brown bits from the pan into the liquid.
\medskip
\item Bring the liquid to a boil, then reduce the heat to medium and gently boil, uncovered, until the sauce is reduced by about half, slightly thickened, and darkened in color, 10 to 15 minutes (you’re going for a thin cream sauce; it won’t start to thicken until the very end of the cooking time). Add the chicken back to the pan, along with any juices that accumulated on the plate. Reduce the heat to low and simmer until the chicken is warmed through and the sauce thickens a bit more, 2 to 3 minutes. Sprinkle with parsley, if using, and serve.
\newline 
\end{enumerate}
\end{multicols}
